# 1.以下代码输出什么

var a = {}
var b = {}
a[b] = 3

答案：

{ '[object Object]': 3 }

# 2.以下代码输出什么

(function(){
    var x = y = 1;
})();
console.log(y);    // 1
console.log(x);    // ReferenceError: x is not defined

答案：

1
ReferenceError: x is not defined

//y 被赋值到全局. x 是局部变量. 所以打印 x 的时候会报 ReferenceError

# 3.以下代码输出什么

var a = 10
var obj = {
  a: 20,
  say: () => {
    console.log(this.a)
  }
}
obj.say() // undefined
var anotherObj = { a:30 }
obj.say.apply(anotherObj) // undefined

答案：

10
10
